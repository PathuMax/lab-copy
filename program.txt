        <input type="text" id="studentName" placeholder="Name" required>

    <button id="testButton">Send Test Student</button>

    <script>
        // ====== PubNub Keys ======
        const publishKey = "pub-c-e1f4c1a6-b73b-4e3a-aa9a-53b9d8f5dcd5";
        const subscribeKey = "sub-c-cac16290-c806-48a0-9d3b-161a9996a665";
        const channel = "STUDENTDASHBOARD";
        const tableBody = document.querySelector('#studentsTable tbody');

        // ====== Initialize PubNub ======
        const pubnub = new PubNub({
            publishKey: publishKey,
            subscribeKey: subscribeKey,
            uuid: "dashboard-" + Math.random() // unique id per tab
        });

        // ====== Add or update student row ======
        function addOrUpdateStudent(student) {
            let existingRow = document.getElementById(student.id);
            if (existingRow) {
                existingRow.innerHTML = `
                    <td>${student.id}</td>
                    <td>${student.name}</td>
                    <td>${student.age}</td>
                    <td>${student.class}</td>
                    <td>${student.grade}</td>
                `;
            } else {
                const row = document.createElement("tr");
                row.id = student.id;
                row.innerHTML = `
                    <td>${student.id}</td>
                    <td>${student.name}</td>
                    <td>${student.age}</td>
                    <td>${student.class}</td>
                    <td>${student.grade}</td>
                `;
                tableBody.appendChild(row);
            }
        }

        // ====== Subscribe to PubNub messages ======
        pubnub.subscribe({ channels: [channel] });
        pubnub.addListener({
            message: function (event) {
                const student = event.message;
                addOrUpdateStudent(student);
            }
        });

        // ====== Send student via PubNub ======
        function sendStudent(student) {
            pubnub.publish({
                channel: channel,
                message: student
            });
        }

        // ====== Test button ======
        document.getElementById("testButton").addEventListener("click", function () {
            const testStudent = {
                id: "TEST001",
                name: "Test Student",
                age: 20,
                class: "IT",
                grade: "A+"
            };
            sendStudent(testStudent);
        });

        // ====== Form submit ======
        document.getElementById("sendForm").addEventListener("submit", function (e) {
            e.preventDefault();
            const student = {
                id: document.getElementById("studentId").value,
                name: document.getElementById("studentName").value,
                age: Number(document.getElementById("studentAge").value),
                class: document.getElementById("studentClass").value,
                grade: document.getElementById("studentGrade").value
            };
            sendStudent(student);
            this.reset();
        });
    </script>
